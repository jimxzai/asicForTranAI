1//! Gazillioner Wallet - Multi-Chain Cold Storage
2//!
3//! This crate provides a secure, air-gapped capable wallet for multiple chains.
4//!
5//! Supported Chains:
6//! - Bitcoin (BTC) - Native SegWit (BIP84)
7//! - Ethereum (ETH) - EIP-155, EIP-1559
8//! - EVM L2s - Polygon, Arbitrum, Base, Optimism
9//! - Solana (SOL) - Ed25519 signing
10//!
11//! Features:
12//! - BIP39 mnemonic generation and recovery (24 words)
13//! - BIP32/BIP44 HD key derivation
14//! - SLIP-0044 coin type standards
15//! - Air-gapped signing via QR codes
16//! - Zeroization of sensitive data
17

46/// Supported networks/chains
47#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize)]
48pub enum Network {
49    // Bitcoin
50    Bitcoin,
51    BitcoinTestnet,
52    // Ethereum L1
53    Ethereum,
54    EthereumGoerli,
55    EthereumSepolia,
56    // EVM L2s
57    Polygon,
58    PolygonMumbai,
59    Arbitrum,
60    ArbitrumGoerli,
61    Base,
62    BaseSepolia,
63    Optimism,
64    OptimismGoerli,
65    // Solana
66    Solana,
67    SolanaDevnet,
68}
69
70impl Network {
71    pub fn is_mainnet(&self) -> bool {
72        matches!(
73            self,
74            Network::Bitcoin
75                | Network::Ethereum
76                | Network::Polygon
77                | Network::Arbitrum
78                | Network::Base
79                | Network::Optimism
80                | Network::Solana
81        )
82    }
83
84    /// BIP44 coin type (SLIP-0044)
85    pub fn coin_type(&self) -> u32 {
86        match self {
87            Network::Bitcoin => 0,
88            Network::BitcoinTestnet => 1,
89            Network::Ethereum
90            | Network::EthereumGoerli
91            | Network::EthereumSepolia
92            | Network::Polygon
93            | Network::PolygonMumbai
94            | Network::Arbitrum
95            | Network::ArbitrumGoerli
96            | Network::Base
97            | Network::BaseSepolia
98            | Network::Optimism
99            | Network::OptimismGoerli => 60,
100            Network::Solana | Network::SolanaDevnet => 501,
101        }
102    }
103
104    /// Chain ID for EVM networks
105    pub fn chain_id(&self) -> Option<u64> {
106        match self {
107            Network::Ethereum => Some(1),
108            Network::EthereumGoerli => Some(5),
109            Network::EthereumSepolia => Some(11155111),
110            Network::Polygon => Some(137),
111            Network::PolygonMumbai => Some(80001),
112            Network::Arbitrum => Some(42161),
113            Network::ArbitrumGoerli => Some(421613),
114            Network::Base => Some(8453),
115            Network::BaseSepolia => Some(84532),
116            Network::Optimism => Some(10),
117            Network::OptimismGoerli => Some(420),
118            _ => None,
119        }
120    }
121
122    /// Whether this is an EVM-compatible chain
123    pub fn is_evm(&self) -> bool {
124        self.chain_id().is_some()
125    }
126
127    /// Whether this is a Bitcoin-based chain
128    pub fn is_bitcoin(&self) -> bool {
129        matches!(self, Network::Bitcoin | Network::BitcoinTestnet)
130    }
131
132    /// Whether this is Solana
133    pub fn is_solana(&self) -> bool {
134        matches!(self, Network::Solana | Network::SolanaDevnet)
135    }
