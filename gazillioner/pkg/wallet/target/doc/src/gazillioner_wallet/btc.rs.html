23/// Bitcoin address types
24#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
25pub enum AddressType {
26    /// Native SegWit (bc1q...) - BIP84
27    NativeSegwit,
28    /// SegWit-compatible (3...) - BIP49
29    SegwitCompat,
30    /// Legacy (1...) - BIP44
31    Legacy,
32    /// Taproot (bc1p...) - BIP86
33    Taproot,
34}
35
36/// Bitcoin address
37#[derive(Debug, Clone, Serialize, Deserialize)]
38pub struct BitcoinAddress {
39    pub address: String,
40    pub address_type: AddressType,
41    pub derivation_path: String,
42    pub index: u32,
43    pub is_change: bool,
44}
45
46impl BitcoinAddress {
47    /// Create from public key (Native SegWit)
48    pub fn from_public_key(
49        pubkey: &secp256k1::PublicKey,
50        network: Network,
51        path: &DerivationPath,
52        index: u32,
53        is_change: bool,
54    ) -> Result<Self> {
55        let btc_network = match network {
56            Network::Bitcoin => BtcNetwork::Bitcoin,
57            Network::BitcoinTestnet => BtcNetwork::Testnet,
58            _ => return Err(WalletError::NetworkMismatch {
59                expected: "Bitcoin".into(),
60                actual: format!("{:?}", network),
61            }),
62        };
63
64        // Create compressed public key
65        let compressed = CompressedPublicKey::from_slice(&pubkey.serialize())
66            .map_err(|e| WalletError::InvalidPublicKey)?;
67
68        // Generate Native SegWit address (P2WPKH)
69        let address = Address::p2wpkh(&compressed, btc_network);
70
71        Ok(Self {
72            address: address.to_string(),
73            address_type: AddressType::NativeSegwit,
74            derivation_path: path.to_string(),
75            index,
76            is_change,
77        })
78    }
79
80    /// Parse address string
81    pub fn parse(address: &str, network: Network) -> Result<Address<bitcoin::address::NetworkUnchecked>> {
82        address.parse()
83            .map_err(|e| WalletError::InvalidAddress(format!("{}", e)))
84    }
85}

106/// Unsigned Bitcoin transaction
107#[derive(Debug, Clone, Serialize, Deserialize)]
108pub struct BitcoinTransaction {
109    pub inputs: Vec<Utxo>,
110    pub outputs: Vec<TxOutput>,
111    pub change_address: Option<String>,
112    pub fee_sats: u64,
113    pub fee_rate_sat_vb: f64,
114}
115
116impl BitcoinTransaction {
117    /// Create a new transaction
118    pub fn new(
119        inputs: Vec<Utxo>,
120        outputs: Vec<TxOutput>,
121        change_address: Option<String>,
122        fee_rate_sat_vb: f64,
123    ) -> Result<Self> {
124        // Calculate total input
125        let total_input: u64 = inputs.iter().map(|u| u.amount_sats).sum();
126
127        // Calculate total output
128        let total_output: u64 = outputs.iter().map(|o| o.amount_sats).sum();
129
130        // Estimate transaction size (simplified)
131        // P2WPKH: ~68 vbytes per input, ~31 vbytes per output, ~10.5 vbytes overhead
132        let estimated_vsize = 10.5 + (inputs.len() as f64 * 68.0) + (outputs.len() as f64 * 31.0);
133        let fee_sats = (estimated_vsize * fee_rate_sat_vb).ceil() as u64;
134
135        if total_input < total_output + fee_sats {
136            return Err(WalletError::InsufficientFunds {
137                needed: total_output + fee_sats,
138                available: total_input,
139            });
140        }
141
142        Ok(Self {
143            inputs,
144            outputs,
145            change_address,
146            fee_sats,
147            fee_rate_sat_vb,
148        })
149    }
150
151    /// Get total input amount
152    pub fn total_input(&self) -> u64 {
153        self.inputs.iter().map(|u| u.amount_sats).sum()
154    }
155
156    /// Get total output amount (excluding change)
157    pub fn total_output(&self) -> u64 {
158        self.outputs.iter().map(|o| o.amount_sats).sum()
159    }
160
161    /// Get change amount
162    pub fn change_amount(&self) -> u64 {
163        self.total_input().saturating_sub(self.total_output() + self.fee_sats)
164    }
165}

177/// Bitcoin wallet for managing addresses and signing
178pub struct BitcoinWallet {
179    master_key: ExtendedPrivateKey,
180    network: Network,
181    next_receive_index: u32,
182    next_change_index: u32,
183}
184
185impl BitcoinWallet {
186    /// Create wallet from master extended private key
187    pub fn new(master_key: ExtendedPrivateKey, network: Network) -> Result<Self> {
188        if !matches!(network, Network::Bitcoin | Network::BitcoinTestnet) {
189            return Err(WalletError::NetworkMismatch {
190                expected: "Bitcoin or BitcoinTestnet".into(),
191                actual: format!("{:?}", network),
192            });
193        }
194
195        Ok(Self {
196            master_key,
197            network,
198            next_receive_index: 0,
199            next_change_index: 0,
200        })
201    }
202
203    /// Generate new receive address
204    pub fn new_receive_address(&mut self) -> Result<BitcoinAddress> {
205        let index = self.next_receive_index;
206        self.next_receive_index += 1;
207        self.get_address(false, index)
208    }
209
210    /// Generate new change address
211    pub fn new_change_address(&mut self) -> Result<BitcoinAddress> {
212        let index = self.next_change_index;
213        self.next_change_index += 1;
214        self.get_address(true, index)
215    }
216
217    /// Get address at specific index
218    pub fn get_address(&self, is_change: bool, index: u32) -> Result<BitcoinAddress> {
219        let path = DerivationPath::bitcoin_native_segwit(0, is_change, index);
220        let keypair = KeyPair::from_xpriv(&self.master_key, &path)?;
221
222        BitcoinAddress::from_public_key(
223            keypair.public_key(),
224            self.network,
225            &path,
226            index,
227            is_change,
228        )
229    }
230
231    /// Get all generated receive addresses
232    pub fn get_all_addresses(&self) -> Vec<BitcoinAddress> {
233        let mut addresses = Vec::new();
234        for i in 0..self.next_receive_index {
235            if let Ok(addr) = self.get_address(false, i) {
236                addresses.push(addr);
237            }
238        }
239        addresses
240    }
241
242    /// Sign a transaction
243    pub fn sign_transaction(&self, tx: &BitcoinTransaction) -> Result<SignedBitcoinTransaction> {
244        let secp = Secp256k1::new();
245
246        let btc_network = match self.network {
247            Network::Bitcoin => BtcNetwork::Bitcoin,
248            Network::BitcoinTestnet => BtcNetwork::Testnet,
249            _ => unreachable!(),
250        };
251
252        // Build unsigned transaction
253        let mut unsigned_tx = Transaction {
254            version: Version::TWO,
255            lock_time: LockTime::ZERO,
256            input: vec![],
257            output: vec![],
258        };
259
260        // Add inputs
261        for utxo in &tx.inputs {
262            let txid: bitcoin::Txid = utxo.txid.parse()
263                .map_err(|e| WalletError::InvalidTransaction(format!("Invalid txid: {}", e)))?;
264
265            unsigned_tx.input.push(TxIn {
266                previous_output: bitcoin::OutPoint {
267                    txid,
268                    vout: utxo.vout,
269                },
270                script_sig: ScriptBuf::new(),
271                sequence: Sequence::ENABLE_RBF_NO_LOCKTIME,
272                witness: Witness::default(),
273            });
274        }
275
276        // Add outputs
277        for output in &tx.outputs {
278            let address: Address<bitcoin::address::NetworkUnchecked> = output.address.parse()
279                .map_err(|e| WalletError::InvalidAddress(format!("{}", e)))?;
280
281            let address = address.assume_checked();
282
283            unsigned_tx.output.push(TxOut {
284                value: bitcoin::Amount::from_sat(output.amount_sats),
285                script_pubkey: address.script_pubkey(),
286            });
287        }
288
289        // Add change output if needed
290        let change_amount = tx.change_amount();
291        if change_amount > 546 { // Dust threshold
292            if let Some(change_addr) = &tx.change_address {
293                let address: Address<bitcoin::address::NetworkUnchecked> = change_addr.parse()
294                    .map_err(|e| WalletError::InvalidAddress(format!("{}", e)))?;
295
296                let address = address.assume_checked();
297
298                unsigned_tx.output.push(TxOut {
299                    value: bitcoin::Amount::from_sat(change_amount),
300                    script_pubkey: address.script_pubkey(),
301                });
302            }
303        }
304
305        // Sign each input
306        let mut sighash_cache = SighashCache::new(&unsigned_tx);
307
308        for (i, utxo) in tx.inputs.iter().enumerate() {
309            // Derive the key for this input
310            let path = DerivationPath::from_str(&utxo.derivation_path)?;
311            let keypair = KeyPair::from_xpriv(&self.master_key, &path)?;
312
313            // Create compressed public key
314            let compressed = CompressedPublicKey::from_slice(&keypair.public_key_bytes())
315                .map_err(|_| WalletError::InvalidPublicKey)?;
316
317            // Create script pubkey for P2WPKH
318            let script_pubkey = ScriptBuf::new_p2wpkh(&compressed.wpubkey_hash());
319
320            // Compute sighash
321            let sighash = sighash_cache.p2wpkh_signature_hash(
322                i,
323                &script_pubkey,
324                bitcoin::Amount::from_sat(utxo.amount_sats),
325                EcdsaSighashType::All,
326            ).map_err(|e| WalletError::SigningFailed(format!("Sighash error: {}", e)))?;
327
328            // Sign
329            let msg = Message::from_digest(sighash.to_byte_array());
330            let sig = secp.sign_ecdsa(&msg, keypair.secret_key());
331
332            // Create Bitcoin signature with sighash type
333            let btc_sig = BtcSignature {
334                signature: sig,
335                sighash_type: EcdsaSighashType::All,
336            };
337
338            // Build witness
339            let mut witness = Witness::new();
340            witness.push(btc_sig.to_vec());
341            witness.push(keypair.public_key_bytes());
342
343            // Update transaction with witness
344            sighash_cache.witness_mut(i).unwrap().clone_from(&witness);
345        }
346
347        let signed_tx = sighash_cache.into_transaction();
348
349        // Serialize
350        let raw_hex = bitcoin::consensus::encode::serialize_hex(&signed_tx);
351        let txid = signed_tx.compute_txid().to_string();
352        let size_bytes = bitcoin::consensus::encode::serialize(&signed_tx).len();
353        let vsize = signed_tx.vsize();
354
355        Ok(SignedBitcoinTransaction {
356            raw_hex,
357            txid,
358            size_bytes,
359            vsize,
360            fee_sats: tx.fee_sats,
361        })
362    }
363
364    /// Get master fingerprint
365    pub fn fingerprint(&self) -> [u8; 4] {
366        self.master_key.to_extended_public_key().fingerprint()
367    }
368
369    /// Get network
370    pub fn network(&self) -> Network {
371        self.network
372    }

375#[cfg(test)]
376mod tests {
377    use super::*;
378    use crate::mnemonic::Mnemonic;
379
380    fn test_wallet() -> BitcoinWallet {
381        let mnemonic = Mnemonic::from_phrase(
382            "abandon abandon abandon abandon abandon abandon abandon abandon abandon abandon abandon about"
383        ).unwrap();
384        let seed = mnemonic.to_seed_no_passphrase();
385        let master = ExtendedPrivateKey::from_seed(&seed, Network::Bitcoin).unwrap();
386        BitcoinWallet::new(master, Network::Bitcoin).unwrap()
387    }
388
389    #[test]
390    fn test_generate_address() {
391        let mut wallet = test_wallet();
392        let addr = wallet.new_receive_address().unwrap();
393
394        // Should be a bc1 address
395        assert!(addr.address.starts_with("bc1q"));
396        assert_eq!(addr.address_type, AddressType::NativeSegwit);
397    }
398
399    #[test]
400    fn test_address_derivation_consistency() {
401        let wallet = test_wallet();
402
403        let addr1 = wallet.get_address(false, 0).unwrap();
404        let addr2 = wallet.get_address(false, 0).unwrap();
405
406        // Same index should produce same address
407        assert_eq!(addr1.address, addr2.address);
408
409        // Different index should produce different address
410        let addr3 = wallet.get_address(false, 1).unwrap();
411        assert_ne!(addr1.address, addr3.address);
412    }
413
414    #[test]
415    fn test_change_vs_receive() {
416        let wallet = test_wallet();
417
418        let receive = wallet.get_address(false, 0).unwrap();
419        let change = wallet.get_address(true, 0).unwrap();
420
421        // Same index but different chains should produce different addresses
422        assert_ne!(receive.address, change.address);
423    }
