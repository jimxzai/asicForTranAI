@app.get("/v1/mining/status")
async def mining_status():
    """Get Antminer S19 status via local API"""
    import httpx

    # Antminer local API (replace with your miner's IP)
    MINER_IP = os.environ.get("MINER_IP", "192.168.1.100")

    try:
        async with httpx.AsyncClient() as client:
            # Get miner stats
            resp = await client.get(
                f"http://{MINER_IP}/cgi-bin/stats.cgi",
                auth=("root", "root"),  # Default credentials
                timeout=5.0
            )
            stats = resp.json()

            hashrate = stats.get("STATS", [{}])[0].get("rate_5s", 0)
            temp = stats.get("STATS", [{}])[0].get("temp", 0)

            return {
                "online": True,
                "hashrate_ths": hashrate / 1e12,
                "temperature_c": temp,
                "power_w": 3250,  # Estimated
                "daily_btc_estimate": hashrate / 1e12 * 0.0000037,  # Rough
                "status": "mining"
            }
    except Exception as e:
        return {
            "online": False,
            "error": str(e)
        }
